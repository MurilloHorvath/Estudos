< (menor que)
>= (maior ou igual que)
<= (menor ou igual que)
== (igual que)

operadores lógicos

! false = true

&& (Conjunção)
true && true = true
true && false = false

false && false = false

|| (Disjunção)
true || true = true
true || false = true
false || true = true
false || false = false
